```javascript
export const useThemeStore = defineStore('theme', {
```

### 2.4 进阶安全机制

#### 2.4.1 连续登录失败处理
```javascript
// stores/auth.js
export const useAuthStore = defineStore('auth', {
  state: () => ({
    loginFailCount: 0,
    needCaptcha: false
  }),
  actions: {
    async login(credentials) {
      try {
        const res = await api.post('/auth/login', {
          ...credentials,
          captcha: this.needCaptcha ? credentials.captcha : undefined
        })
        
        // 登录成功，重置失败计数
        this.loginFailCount = 0
        this.needCaptcha = false
        this.token = res.data.token
        
      } catch (error) {
        this.loginFailCount++
        
        // 连续失败5次，启用图形验证码
        if (this.loginFailCount >= 5) {
          this.needCaptcha = true
          ElMessage.warning('连续登录失败，请输入验证码')
        }
        
        throw error
      }
    }
  }
})
```

### 3.5 拖拽上传食物照片

```vue
<template>
  <div 
    class="upload-area" 
    @dragover.prevent="isDragging = true"
    @dragleave.prevent="isDragging = false"
    @drop.prevent="handleDrop"
    :class="{ 'drag-active': isDragging }">
    
    <input 
      type="file" 
      ref="fileInput" 
      hidden 
      accept="image/*"
      @change="handleFileSelect" />
    
    <div v-if="!previewUrl" class="upload-prompt">
      <i class="upload-icon">📸</i>
      <p>拖拽食物照片到这里</p>
      <p class="or-text">或</p>
      <el-button type="primary" @click="triggerUpload">点击上传</el-button>
    </div>
    
    <div v-else class="preview-container">
      <img :src="previewUrl" alt="食物预览" />
      <div class="preview-actions">
        <el-button @click="recognizeFood" type="success" :loading="recognizing">
          AI识别食物
        </el-button>
        <el-button @click="clearPreview" type="danger">重新上传</el-button>
      </div>
      
      <div v-if="recognitionResult" class="recognition-result">
        <h4>识别结果：</h4>
        <p>食物名称：{{ recognitionResult.foodName }}</p>
        <p>预估重量：{{ recognitionResult.weight }}克</p>
        <p>热量：{{ recognitionResult.calories }}kcal</p>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { ElMessage } from 'element-plus'
import api from '@/services/api'

const isDragging = ref(false)
const previewUrl = ref(null)
const fileInput = ref(null)
const recognizing = ref(false)
const recognitionResult = ref(null)

const handleDrop = (e) => {
  isDragging.value = false
  const file = e.dataTransfer.files[0]
  processFile(file)
}

const handleFileSelect = (e) => {
  const file = e.target.files[0]
  processFile(file)
}

const processFile = (file) => {
  if (!file.type.startsWith('image/')) {
    return ElMessage.error('仅支持图片格式')
  }
  
  if (file.size > 5 * 1024 * 1024) {
    return ElMessage.error('图片大小不能超过5MB')
  }
  
  const reader = new FileReader()
  reader.onload = (e) => {
    previewUrl.value = e.target.result
  }
  reader.readAsDataURL(file)
}

const recognizeFood = async () => {
  recognizing.value = true
  try {
    const res = await api.post('/ai/recognize-food', {
      image: previewUrl.value
    })
    recognitionResult.value = res.data
    ElMessage.success('识别成功！')
  } catch (error) {
    ElMessage.error('识别失败，请重试')
  } finally {
    recognizing.value = false
  }
}

const triggerUpload = () => {
  fileInput.value.click()
}

const clearPreview = () => {
  previewUrl.value = null
  recognitionResult.value = null
}
</script>
```

### 3.7 数据缓存策略

```javascript
// services/cache.js
export class CacheService {
  constructor(namespace = 'nutriai') {
    this.namespace = namespace
    this.cacheExpiry = 24 * 60 * 60 * 1000 // 24小时
  }
  
  set(key, data) {
    const cacheData = {
      data,
      timestamp: Date.now()
    }
    localStorage.setItem(
      `${this.namespace}_${key}`, 
      JSON.stringify(cacheData)
    )
  }
  
  get(key) {
    const cached = localStorage.getItem(`${this.namespace}_${key}`)
    if (!cached) return null
    
    const { data, timestamp } = JSON.parse(cached)
    
    // 检查是否过期
    if (Date.now() - timestamp > this.cacheExpiry) {
      this.remove(key)
      return null
    }
    
    return data
  }
  
  remove(key) {
    localStorage.removeItem(`${this.namespace}_${key}`)
  }
}

// 使用示例
const cache = new CacheService()

// 获取用户资料时先检查缓存
async function getUserProfile(userId) {
  const cached = cache.get(`user_profile_${userId}`)
  if (cached) {
    console.log('使用缓存数据')
    return cached
  }
  
  const res = await api.get(`/user/profile/${userId}`)
  cache.set(`user_profile_${userId}`, res.data)
  return res.data
}
```

### 4.4 虚拟滚动实现（权益列表）

当权益条目超过100个时，使用虚拟滚动避免性能问题：

```vue
<template>
  <div 
    class="virtual-list-container" 
    ref="containerRef"
    @scroll="handleScroll"
    :style="{ height: containerHeight + 'px' }">
    
    <!-- 占位元素，撑起总高度 -->
    <div :style="{ height: totalHeight + 'px', position: 'relative' }">
      <!-- 实际渲染的可见元素 -->
      <div 
        class="virtual-list-content"
        :style="{ transform: `translateY(${offsetY}px)` }">
        <div 
          v-for="item in visibleItems" 
          :key="item.id"
          class="privilege-item"
          :style="{ height: itemHeight + 'px' }">
          <i :class="item.icon"></i>
          <div class="privilege-info">
            <h4>{{ item.title }}</h4>
            <p>{{ item.description }}</p>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue'

const props = defineProps({
  items: {
    type: Array,
    required: true
  }
})

const containerRef = ref(null)
const containerHeight = ref(600) // 容器高度
const itemHeight = 80 // 每项高度
const bufferSize = 5 // 缓冲区大小

const scrollTop = ref(0)

// 计算总高度
const totalHeight = computed(() => props.items.length * itemHeight)

// 计算可见区域起始索引
const startIndex = computed(() => {
  return Math.max(0, Math.floor(scrollTop.value / itemHeight) - bufferSize)
})

// 计算可见区域结束索引
const endIndex = computed(() => {
  const visibleCount = Math.ceil(containerHeight.value / itemHeight)
  return Math.min(
    props.items.length,
    startIndex.value + visibleCount + bufferSize * 2
  )
})

// 可见的元素列表
const visibleItems = computed(() => {
  return props.items.slice(startIndex.value, endIndex.value)
})

// 偏移量
const offsetY = computed(() => startIndex.value * itemHeight)

const handleScroll = (e) => {
  scrollTop.value = e.target.scrollTop
}

onMounted(() => {
  // 可选：动态获取容器高度
  if (containerRef.value) {
    containerHeight.value = containerRef.value.clientHeight
  }
})
</script>
```
