#### 3.3.1 组合式函数（Composables）

```javascript
// composables/useDebounce.js
import { ref } from 'vue'

export function useDebounce(fn, delay = 300) {
  let timeoutId = null
  
  return (...args) => {
    if (timeoutId) {
      clearTimeout(timeoutId)
    }
    
    timeoutId = setTimeout(() => {
      fn(...args)
    }, delay)
  }
}

// 使用
import { useDebounce } from '@/composables/useDebounce'

const searchInput = ref('')
const debouncedSearch = useDebounce((value) => {
  console.log('搜索:', value)
}, 500)

watch(searchInput, debouncedSearch)
```

#### 4.1.2 Store编写规范

```javascript
// stores/auth.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import router from '@/router'
import api from '@/services/api'

export const useAuthStore = defineStore('auth', () => {
  // State（使用ref）
  const token = ref(localStorage.getItem('nutri_token') || null)
  const userInfo = ref(JSON.parse(localStorage.getItem('user_info')) || null)
  const loginFailCount = ref(0)
  
  // Getters（使用computed）
  const isLoggedIn = computed(() => !!token.value)
  const userName = computed(() => userInfo.value?.nickname || '游客')
  const userRole = computed(() => userInfo.value?.role || 'guest')
  
  // Actions（普通函数）
  async function login(credentials) {
    try {
      const res = await api.post('/auth/login', credentials)
      
      token.value = res.data.token
      userInfo.value = res.data.user
      
      // 持久化存储
      localStorage.setItem('nutri_token', res.data.token)
      localStorage.setItem('user_info', JSON.stringify(res.data.user))
      
      // 重置失败计数
      loginFailCount.value = 0
      
      return res.data
    } catch (error) {
      loginFailCount.value++
      throw error
    }
  }
  
  function logout() {
    token.value = null
    userInfo.value = null
    localStorage.removeItem('nutri_token')
    localStorage.removeItem('user_info')
    router.push('/login')
  }
  
  function hasRole(role) {
    const roleHierarchy = {
      guest: 0,
      user: 1,
      operator: 2,
      admin: 3,
      super_admin: 4
    }
    
    return roleHierarchy[userRole.value] >= roleHierarchy[role]
  }
  
  return {
    // State
    token,
    userInfo,
    loginFailCount,
    // Getters
    isLoggedIn,
    userName,
    userRole,
    // Actions
    login,
    logout,
    hasRole
  }
})
```

### 4.2 Store使用规范

```vue
<script setup>
import { useAuthStore } from '@/stores/auth'
import { storeToRefs } from 'pinia'

const authStore = useAuthStore()

// 解构state和getters时使用storeToRefs保持响应性

// Actions可以直接解构
```

### 5.3 路由守卫实现

```javascript
// router/guards/index.js
import { useAuthStore } from '@/stores/auth'
import { ElMessage } from 'element-plus'

export function setupRouterGuards(router) {
  // 全局前置守卫
  router.beforeEach((to, from, next) => {
    const authStore = useAuthStore()
    
    // 设置页面标题
    document.title = to.meta.title 
      ? `${to.meta.title} - AI健康饮食规划助手`
      : 'AI健康饮食规划助手'
    
    // 认证检查
    if (to.meta.requiresAuth && !authStore.isLoggedIn) {
      ElMessage.warning('请先登录')
      next({
        path: '/login',
        query: { redirect: to.fullPath }
      })
      return
    }
    
    // 权限检查
    if (to.meta.role && !authStore.hasRole(to.meta.role)) {
      ElMessage.error('权限不足')
      next(false)
      return
    }
    
    next()
  })
  
  // 全局后置钩子
  router.afterEach((to, from) => {
    // 页面访问统计
    if (window._hmt) {
      window._hmt.push(['_trackPageview', to.fullPath])
    }
  })
}
```

### 9.3 常见问题准备

#### 9.3.1 技术选型类
**Q: 为什么选择Pinia而不是Vuex？**  
A: Pinia是Vue官方推荐的新一代状态管理工具，相比Vuex具有以下优势：
1. 更好的TypeScript支持
2. 去除了mutations，简化代码
3. 模块化设计，无需命名空间
4. 更小的打包体积（约1KB）

**Q: 为什么使用WebSocket而不是轮询？**  
A: WebSocket提供全双工通信，相比HTTP轮询：
1. 实时性更好（延迟<100ms vs 轮询间隔通常≥1s）
2. 服务器资源消耗更少（减少80%请求数）
3. 更适合聊天场景的流式输出

#### 9.3.2 实现细节类
**Q: 如何实现AI回复的打字效果？**  
A: 通过WebSocket分块接收AI响应，前端使用setTimeout模拟打字：
```javascript
const chunk = response.substring(i, i + chunkSize)
setTimeout(() => {
  message.content += chunk
}, i * 50) // 每个字符延迟50ms
```

**Q: 虚拟滚动的原理是什么？**  
A: 只渲染可见区域的DOM元素：
1. 计算可见区域起始索引：`Math.floor(scrollTop / itemHeight)`
2. 仅渲染可见数量的元素（通常20-30个）
3. 使用transform偏移实现滚动效果
4. 大幅减少DOM数量（1000+条目降至30个DOM节点）

#### 9.3.3 问题解决类
**Q: 开发中遇到的最大困难是什么？**  
A: （示例回答）最大困难是AI响应的流式输出实现。
- **问题**：通义千问API返回完整响应，无法实现打字效果
- **解决方案**：后端使用CompletableFuture异步处理，将完整响应分块通过WebSocket发送
- **效果**：用户体验提升40%（用户调研反馈）
